### void 타입 
리턴값이 없을 댸 

function double (a, b) {
    console.log(a,b)
}

언제사용할까 함수의 타입을 지정할 때 리턴값이 없을경우 주로 사용

interface NewType {
    name:string;
    age:number;
    double : (a:number, b:number) => void 형식으로 작성
    getAge:()=>number // 리턴값이 있을경우
    getAge:()=>void // 리턴값이 없을경우
}

type AorB = {a:number,b?:never} | {a?:never,b:number}
